    Queries 

    self.db['entry'].aggregate(
  [{
    '$match': {
      'author_id': {
        '$ne': None,
        '$exists': True
       }
     }
    },
    {
      '$lookup': {
        'from': 'author',
        'localField': 'author_id',
        'foreignField': 'id',
        'as': 'author'
      }
    },
    {
      '$unwind': '$author'
    },
    {
      '$lookup': {
        'from': 'blog',
        'localField': 'blog_id',
        'foreignField': 'id',
        'as': 'blog'
      }
    },
    {
      '$unwind': {
        'path': '$blog',
        'preserveNullAndEmptyArrays': True
      }
    },
    {
      '$addFields': {
        'blog': {
          '$ifNull': ['$blog', {
            'id': None,
            'title': None
          }]
        }
      }
    },
    {
      '$match': {
        'author.name': {
          '$eq': 'Paul'
        }
      }
    }, 
    {
      '$project': {
        'id': True,
        'blog_id': True,
        'author_id': True,
        'content': True,
        'blog.id': True,
        'blog.title': True
      }
    }]

    qs = Entry.objects.filter(author__name='Paul').select_related('blog')
